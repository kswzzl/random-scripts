--- STEP 0: Create the unit namespace ----------------------------------------

kubectl create namespace unit-UNITNAME
kubectl label namespace unit-UNITNAME agent-fleet/unit=UNITNAME


--- STEP 1: Default deny all -------------------------------------------------
--- Every pod gets zero networking. This is the foundation.

kubectl apply -n unit-UNITNAME -f - << 'EOF'
apiVersion: networking.k8s.io/v1
kind: NetworkPolicy
metadata:
  name: default-deny-all
spec:
  podSelector: {}
  policyTypes:
  - Ingress
  - Egress
EOF


--- STEP 2: Allow DNS for agents ----------------------------------------------
--- Without this, pods can't resolve service names like squid.unit-UNITNAME.svc

apiVersion: cilium.io/v2
kind: CiliumNetworkPolicy
metadata:
  name: agent-dns-restrict
spec:
  endpointSelector:
    matchLabels:
      role: agent
  egress:
  - toEndpoints:
    - matchLabels:
        k8s-app: kube-dns
        k8s:io.kubernetes.pod.namespace: kube-system
    toPorts:
    - ports:
      - port: "53"
        protocol: ANY
      rules:
        dns:
        - matchPattern: "*.svc.cluster.local"
        - matchPattern: "www.google.com"


--- STEP 3: Allow agents to reach squid only ----------------------------------
--- Agents can connect to squid on 3128. Nothing else.

kubectl apply -n unit-UNITNAME -f - << 'EOF'
apiVersion: networking.k8s.io/v1
kind: NetworkPolicy
metadata:
  name: agent-to-squid
spec:
  podSelector:
    matchLabels:
      role: agent
  policyTypes:
  - Egress
  egress:
  - to:
    - podSelector:
        matchLabels:
          role: proxy
    ports:
    - port: 3128
      protocol: TCP
EOF


--- STEP 4: Allow agents to reach MCP only ------------------------------------
--- Agents can connect to MCP server on 8080. Nothing else.

kubectl apply -n unit-UNITNAME -f - << 'EOF'
apiVersion: networking.k8s.io/v1
kind: NetworkPolicy
metadata:
  name: agent-to-mcp
spec:
  podSelector:
    matchLabels:
      role: agent
  policyTypes:
  - Egress
  egress:
  - to:
    - podSelector:
        matchLabels:
          role: mcp
    ports:
    - port: 8080
      protocol: TCP
EOF


--- STEP 5: Squid policy ------------------------------------------------------
--- Squid accepts inbound from agents only.
--- Squid can reach the internet (it does its own domain filtering).

kubectl apply -n unit-UNITNAME -f - << 'EOF'
apiVersion: networking.k8s.io/v1
kind: NetworkPolicy
metadata:
  name: squid-policy
spec:
  podSelector:
    matchLabels:
      role: proxy
  policyTypes:
  - Ingress
  - Egress
  ingress:
  - from:
    - podSelector:
        matchLabels:
          role: agent
    ports:
    - port: 3128
      protocol: TCP
  egress:
  - {}
EOF


--- STEP 6: MCP policy --------------------------------------------------------
--- MCP accepts inbound from agents only.
--- MCP can resolve DNS and reach squid (for outbound tool calls).

kubectl apply -n unit-UNITNAME -f - << 'EOF'
apiVersion: networking.k8s.io/v1
kind: NetworkPolicy
metadata:
  name: mcp-policy
spec:
  podSelector:
    matchLabels:
      role: mcp
  policyTypes:
  - Ingress
  - Egress
  ingress:
  - from:
    - podSelector:
        matchLabels:
          role: agent
    ports:
    - port: 8080
      protocol: TCP
  egress:
  - to:
    - namespaceSelector: {}
      podSelector:
        matchLabels:
          k8s-app: kube-dns
    ports:
    - port: 53
      protocol: UDP
    - port: 53
      protocol: TCP
  - to:
    - podSelector:
        matchLabels:
          role: proxy
    ports:
    - port: 3128
      protocol: TCP
EOF


--- VERIFICATION ---------------------------------------------------------------
--- Run these after applying all policies to confirm everything works.

# Check all policies are in place
kubectl get networkpolicy -n unit-UNITNAME

# Expected output:
#   NAME               POD-SELECTOR   AGE
#   default-deny-all   <none>         ...
#   agent-allow-dns    role=agent     ...
#   agent-to-squid     role=agent     ...
#   agent-to-mcp       role=agent     ...
#   squid-policy       role=proxy     ...
#   mcp-policy         role=mcp       ...

# Deploy a test agent pod
kubectl apply -n unit-UNITNAME -f - << 'EOF'
apiVersion: v1
kind: Pod
metadata:
  name: net-test
  labels:
    role: agent
spec:
  runtimeClassName: kata-clh
  restartPolicy: Never
  containers:
  - name: test
    image: busybox:1.36
    command: ["sleep", "3600"]
EOF
kubectl wait --for=condition=Ready -n unit-UNITNAME pod/net-test --timeout=60s

# TEST 1: DNS should work
kubectl exec -n unit-UNITNAME net-test -- nslookup google.com
# Expected: resolves successfully

# TEST 2: Direct internet should be blocked
kubectl exec -n unit-UNITNAME net-test -- wget -qO- --timeout=5 http://1.1.1.1
# Expected: timeout / connection refused

# TEST 3: Direct internet by domain should be blocked
kubectl exec -n unit-UNITNAME net-test -- wget -qO- --timeout=5 http://google.com
# Expected: timeout / connection refused

# TEST 4: Squid should be reachable
kubectl exec -n unit-UNITNAME net-test -- wget -qO- --timeout=5 http://squid.unit-UNITNAME.svc.cluster.local:3128
# Expected: squid error page (400 bad request or similar — that's fine, means squid responded)

# TEST 5: K8s API server should be blocked
kubectl exec -n unit-UNITNAME net-test -- wget -qO- --timeout=5 https://kubernetes.default.svc.cluster.local
# Expected: timeout / connection refused

# TEST 6: Cross-namespace should be blocked (if other namespaces exist)
kubectl exec -n unit-UNITNAME net-test -- wget -qO- --timeout=5 http://squid.unit-other.svc.cluster.local:3128
# Expected: timeout / connection refused

# Clean up test pod
kubectl delete -n unit-UNITNAME pod net-test


--- TRAFFIC FLOW SUMMARY ------------------------------------------------------

  agent pod
    CAN  --> squid:3128          (proxy for internet access)
    CAN  --> mcp-server:8080     (tool server)
    CAN  --> kube-dns:53         (name resolution)
    DENY --> everything else     (internet, k8s API, other namespaces)

  squid pod
    ACCEPTS <-- agent pods on 3128
    CAN     --> internet         (filtered by squid.conf allowlist)
    DENY    <-- everything else

  mcp pod
    ACCEPTS <-- agent pods on 8080
    CAN     --> kube-dns:53      (name resolution)
    CAN     --> squid:3128       (outbound tool calls via proxy)
    DENY    <-- everything else


--- REQUIRED POD LABELS --------------------------------------------------------
--- These labels MUST be on your pods for the policies to match.
--- If a pod is missing its label, it falls through to default-deny-all.

  Agent pods:     role: agent
  Squid pods:     role: proxy
  MCP pods:       role: mcp


--- CLEANUP (destroy the whole unit) ------------------------------------------

kubectl delete namespace unit-UNITNAME
